CREATE TABLE customer
(
	id int NOT NULL UNIQUE,
	f_name varchar(32),
	l_name varchar(32) NOT NULL,
	customer_role varchar(32),
	item_limit int,
	acct_balance float,
	fine_rate float,
	PRIMARY KEY (id)
);

CREATE TABLE employee
(
	id int NOT NULL UNIQUE,
	f_name varchar(32),
	l_name varchar(32) NOT NULL,
	ssn int UNIQUE,
	birthdate date,
	salary float,
	job_title varchar(32),
	phone_no varchar(32),
	PRIMARY KEY (id)
);

CREATE TABLE item (
    id int NOT NULL UNIQUE,
    PRIMARY KEY (id)
);

CREATE TABLE checkoutItem (
    id int NOT NULL UNIQUE,
    item int NOT NULL,
    checkout_date datetime NOT NULL,
    due_date datetime NOT NULL,
    return_date datetime,
    returned bool,
    borrower_id int NOT NULL,
    employee_id int NOT NULL,
    PRIMARY KEY (id),
    FOREIGN KEY (item) REFERENCES item(id),
    FOREIGN KEY (borrower_id) REFERENCES customer(id),
    FOREIGN KEY (employee_id) REFERENCES employee(id)
);

CREATE TABLE location
(
    id int NOT NULL UNIQUE,
    location_name varchar(128) NOT NULL,
    address varchar(128),
    phone_no varchar(32),
    PRIMARY KEY (id)
);

CREATE TABLE author (
    id int NOT NULL UNIQUE,
    f_name varchar(32),
    l_name varchar(32) NOT NULL,
    origin varchar(128),
    author_born int,
    author_died int,
    PRIMARY KEY (id)
);

CREATE TABLE publisher (
    id int NOT NULL UNIQUE,
    publisher_name varchar(32) NOT NULL,
    headquarters varchar(32),
    PRIMARY KEY (id)
);

CREATE TABLE narrator
(
    id int NOT NULL UNIQUE,
    f_name varchar(32),
    l_name varchar(32) NOT NULL,
    origin varchar(128),
    narrator_born int,
    narrator_died int,
    PRIMARY KEY (id)
);

CREATE TABLE series (
    id int NOT NULL UNIQUE,
    series_name varchar(128) NOT NULL,
    total_series int NOT NULL,
    series_number int NOT NULL,
    PRIMARY KEY (id)
);

CREATE TABLE lateFine (
    id int NOT NULL UNIQUE,
    checkout_item int NOT NULL,
    borrower int NOT NULL,
    days_late int,
    fine_amount float,
    paid boolean,
    PRIMARY KEY (id),
    FOREIGN KEY (checkout_item) REFERENCES checkoutItem(id),
    FOREIGN KEY (borrower) REFERENCES customer(id)
);

CREATE TABLE device
 (
    id int NOT NULL UNIQUE,
    device_type varchar(32),
    model varchar(32),
    checked_out boolean,
    waitlist_capacity int,
    location int,
    PRIMARY KEY (id),
    FOREIGN KEY (id) REFERENCES item (id),
    FOREIGN KEY (location) REFERENCES location(id)
);

CREATE TABLE itemRequest (
    req_number int NOT NULL UNIQUE,
    item_id int NOT NULL,
    req_date datetime NOT NULL,
    requester_id int NOT NULL,
    PRIMARY KEY (req_number),
    FOREIGN KEY (item_id) REFERENCES item(id),
    FOREIGN KEY (requester_id) REFERENCES customer(id)
);

CREATE TABLE book (
    id int NOT NULL UNIQUE,
    isbn int,
    title varchar(128) NOT NULL,
    author_id int NOT NULL,
    publisher int,
    publication_year int,
    edition int,
    series int,
    series_position int,
    genre varchar(32) NOT NULL,
    checked_out boolean NOT NULL,
    ebook boolean NOT NULL,
    waitlist_capacity int NOT NULL,
    location int NOT NULL,
    FOREIGN KEY (location) REFERENCES location (id),
    FOREIGN KEY (series) REFERENCES series (id),
    FOREIGN KEY (publisher) REFERENCES publisher (id),
    FOREIGN KEY (author_id) REFERENCES author (id),
    FOREIGN KEY (id) REFERENCES item (id),
    PRIMARY KEY (id)
);

CREATE TABLE dvd
(
    id int NOT NULL UNIQUE,
    title varchar(128) NOT NULL,
    release_date date,
    director varchar(64),
    studio varchar(32),
    location int NOT NULL,
    checked_out boolean NOT NULL,
    FOREIGN KEY (location) REFERENCES location (id),
    FOREIGN KEY (id) REFERENCES item (id),
    PRIMARY KEY (id)
);

CREATE TABLE magazine
(
    id int NOT NULL UNIQUE,
    title varchar(32) NOT NULL,
    issue int,
    issue_date date,
    topic varchar(32),
    location int NOT NULL,
    checked_out boolean NOT NULL,
    FOREIGN KEY (location) REFERENCES location (id),
    FOREIGN KEY (id) REFERENCES item (id),
    PRIMARY KEY (id)
);

CREATE TABLE audiobook
(
    id int NOT NULL UNIQUE,
    isbn int,
    title varchar(128) NOT NULL,
    author_id int NOT NULL,
    narrator_id int NOT NULL,
    publisher int NOT NULL,
    publication_year int,
    edition int,
    series int,
    series_position int,
    genre varchar(32),
    checked_out boolean NOT NULL,
    waitlist_capacity int NOT NULL,
    location int NOT NULL,
    PRIMARY KEY (id),
    FOREIGN KEY (author_id) REFERENCES author(id),
    FOREIGN KEY (narrator_id) REFERENCES narrator(id),
    FOREIGN KEY (publisher) REFERENCES publisher(id),
    FOREIGN KEY (series) REFERENCES series(id),
    FOREIGN KEY (location) REFERENCES location(id)
);
